## 함수 선언하기

```javascript
*예시

 
const sum = function (a, b) {
        // a, b : 인자 (parameter or argument 라고 한다)
        let result = a + b;
        result = result - 1
        result = result * 100 * a

        return result; //꼭 끝에 리턴을 해야함

        // return a+b 간단하게 쓸 경우
    }


*예시2
 const compare = function (a, b) {
        let result;
        if (a == b) {
            result = "같다"
        }
        else if (a > b) {
            result = "크다"
        }
        else {
            result = "작다"
        }

        // 먼저 return 할 값 생각하고.
        // return을 어떤 형식으로 할지 정한다.
        // "같다" "크다" "작다" or true / false or 1 /2 /3

        return result;
    }

```
